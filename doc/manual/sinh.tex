\section{sinh}
sinh --  Performs a hyperbolic sine function. \subsection*{Description}


  Returns the hyperbolic sine of \emph{x}
 (\emph{x}
 in radians). 
\subsection*{Syntax}


 \textbf{sinh}
(x) (no rate restriction)
\subsection*{Examples}


  Here is an example of the sinh opcode. It uses the files \emph{sinh.orc}
 and \emph{sinh.sco}
. 


 \textbf{Example 1. Example of the sinh opcode.}

\begin{lstlisting}
/* sinh.orc */
; Initialize the global variables.
sr = 44100
kr = 4410
ksmps = 10
nchnls = 1

; Instrument #1.
instr 1
  irad = 1
  i1 = sinh(irad)

  print i1
endin
/* sinh.orc */
        
\end{lstlisting}
\begin{lstlisting}
/* sinh.sco */
; Play Instrument #1 for one second.
i 1 0 1
e
/* sinh.sco */
        
\end{lstlisting}
 Its output should a line like this: \begin{lstlisting}
instr 1:  i1 = 1.175
      
\end{lstlisting}

\section{tanh}
tanh --  Performs a hyperbolic tangent function. \subsection*{Description}


  Returns the hyperbolic tangent of \emph{x}
 (\emph{x}
 in radians). 
\subsection*{Syntax}


 \textbf{tanh}
(x) (no rate restriction)
\subsection*{Examples}


  Here is an example of the tanh opcode. It uses the files \emph{tanh.orc}
 and \emph{tanh.sco}
. 


 \textbf{Example 1. Example of the tanh opcode.}

\begin{lstlisting}
/* tanh.orc */
; Initialize the global variables.
sr = 44100
kr = 4410
ksmps = 10
nchnls = 1

; Instrument #1.
instr 1
  irad = 1
  i1 = tanh(irad)

  print i1
endin
/* tanh.orc */
        
\end{lstlisting}
\begin{lstlisting}
/* tanh.sco */
; Play Instrument #1 for one second.
i 1 0 1
e
/* tanh.sco */
        
\end{lstlisting}
 Its output should include a line like this: \begin{lstlisting}
instr 1:  i1 = 0.762
      
\end{lstlisting}

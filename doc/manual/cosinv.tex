\section{cosinv}
cosinv --  Performs a arccosine function. \subsection*{Description}


  Returns the arccosine of \emph{x}
 (\emph{x}
 in radians). 
\subsection*{Syntax}


 \textbf{cosinv}
(x) (no rate restriction)
\subsection*{Examples}


  Here is an example of the cosinv opcode. It uses the files \emph{cosinv.orc}
 and \emph{cosinv.sco}
. 


 \textbf{Example 1. Example of the cosinv opcode.}

\begin{lstlisting}
/* cosinv.orc */
; Initialize the global variables.
sr = 44100
kr = 4410
ksmps = 10
nchnls = 1

; Instrument #1.
instr 1
  irad = 0.5
  i1 = cosinv(irad)

  print i1
endin
/* cosinv.orc */
        
\end{lstlisting}
\begin{lstlisting}
/* cosinv.sco */
; Play Instrument #1 for one second.
i 1 0 1
e
/* cosinv.sco */
        
\end{lstlisting}
 Its output should include lines like this: \begin{lstlisting}
instr 1:  i1 = 1.047
      
\end{lstlisting}
